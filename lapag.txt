<script>

        document.addEventListener("DOMContentLoaded", function() {
        // Retrieve current user from localStorage
        const currentUser = JSON.parse(localStorage.getItem("currentUser"));

        if (!currentUser) {
            // If no current user, redirect to the login page
            window.location.href = "opportUnity_login.html";
        } else {
            // Display the user's first name if logged in
            document.getElementById("userName").textContent = `Welcome, ${currentUser.firstName}!`;

            // Get the list of posted jobs from localStorage
            const jobs = JSON.parse(localStorage.getItem("postedJobs")) || [];
            const postedJobContainer = document.getElementById("posted-job-container");
            const jobListContainer = document.getElementById("job-list-container");

            if (currentUser.role === "employer") {
                // Filter and display jobs posted by the current employer
                const employerJobs = jobs.filter(job => job.employerUsername === currentUser.email); // Use email to identify employer

                if (employerJobs.length > 0) {
                    employerJobs.forEach(function(job) {
                        const jobDiv = document.createElement("div");
                        jobDiv.classList.add("job-post");

                        const jobTitle = document.createElement("h3");
                        jobTitle.textContent = job.jobTitle;
                        const companyName = document.createElement("p");
                        companyName.textContent = `Company: ${job.companyName}`;
                        const jobLocation = document.createElement("p");
                        jobLocation.textContent = `Location: ${job.jobLocation}`;
                        const jobDescription = document.createElement("p");
                        jobDescription.textContent = `Description: ${job.jobDescription}`;
                        const salary = document.createElement("p");
                        salary.textContent = `Salary: ${job.salary}`;

                        jobDiv.appendChild(jobTitle);
                        jobDiv.appendChild(companyName);
                        jobDiv.appendChild(jobLocation);
                        jobDiv.appendChild(jobDescription);
                        jobDiv.appendChild(salary);

                        postedJobContainer.appendChild(jobDiv);
                    });
                } else {
                    postedJobContainer.innerHTML = "<p>You haven't posted any jobs yet.</p>";
                }
            }

            if (currentUser.role === "employee") {
                // If the user is an employee, display jobs posted by employers
                const availableJobs = jobs.filter(job => job.employerUsername !== currentUser.email); // Ensure we exclude the logged-in user

                if (availableJobs.length > 0) {
                    availableJobs.forEach(function(job) {
                        const jobDiv = document.createElement("div");
                        jobDiv.classList.add("job-post");

                        const jobTitle = document.createElement("h3");
                        jobTitle.textContent = job.jobTitle;
                        const companyName = document.createElement("p");
                        companyName.textContent = `Company: ${job.companyName}`;
                        const jobLocation = document.createElement("p");
                        jobLocation.textContent = `Location: ${job.jobLocation}`;
                        const jobDescription = document.createElement("p");
                        jobDescription.textContent = `Description: ${job.jobDescription}`;
                        const salary = document.createElement("p");
                        salary.textContent = `Salary: ${job.salary}`;

                        jobDiv.appendChild(jobTitle);
                        jobDiv.appendChild(companyName);
                        jobDiv.appendChild(jobLocation);
                        jobDiv.appendChild(jobDescription);
                        jobDiv.appendChild(salary);

                        jobListContainer.appendChild(jobDiv);
                    });
                } else {
                    jobListContainer.innerHTML = "<p>No jobs available.</p>";
                }
            }
        }

    <script>

         // Function to handle role selection and button highlighting
    function selectRole(role) {
        const employeeBtn = document.getElementById("employee-btn");
        const employerBtn = document.getElementById("employer-btn");
        const submitBtn = document.getElementById("submit-btn");

        if (role === "employee") {
            employeeBtn.classList.add("active");
            employerBtn.classList.remove("active");
        } else if (role === "employer") {
            employerBtn.classList.add("active");
            employeeBtn.classList.remove("active");
        }

        // Enable submit button once a role is selected
        submitBtn.disabled = false;
    }

    // Function to toggle password visibility
    function togglePasswordVisibility(inputId, button) {
        const input = document.getElementById(inputId);

        if (input.type === "password") {
            input.type = "text";
            button.textContent = "Hide";
        } else {
            input.type = "password";
            button.textContent = "Show";
        }
    }

    // Password validation on confirm password input
    document.getElementById("confirm_password").addEventListener("input", function () {
        const password = document.getElementById("password").value;
        const confirmPassword = this.value;

        if (confirmPassword && password !== confirmPassword) {
            this.setCustomValidity("Passwords do not match");
        } else {
            this.setCustomValidity(""); // Reset validation message
        }
    });

    // Handle the form submission and save user data
    function redirectToLogin(event) {
        event.preventDefault(); // Prevent the default form submission

        let selectedRole = "";
        const role = document.querySelector(".role-btn.active");
        
        // Check if the user has selected a role
        if (!role) {
            alert("Please select a role before creating an account.");
            return; // Prevent form submission
        }

        // Determine selected role
        selectedRole = role.id === "employee-btn" ? "employee" : "employer";

        const firstName = document.getElementById("first_name").value;
        const lastName = document.getElementById("last_name").value;
        const email = document.getElementById("email").value;
        const password = document.getElementById("password").value;

        // Create a user object with the entered data
        const user = {
            firstName,
            lastName,
            email,
            password,
            role: selectedRole
        };

        // Retrieve existing users from localStorage or initialize empty array
        let users = JSON.parse(localStorage.getItem("users")) || [];

        // Add new user to the users array
        users.push(user);

        // Save updated users array to localStorage
        localStorage.setItem("users", JSON.stringify(users));

        // Show success message and redirect to login page
        alert("Sign up successful!");
        window.location.href = "opportUnity_login.html";
    }
